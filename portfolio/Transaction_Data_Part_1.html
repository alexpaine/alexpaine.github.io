Transaction Data - Part I: Cleaning and EDA¶

Transaction level data is invaluable to businesses looking to understand revenue. This series will take a take a close look at the Online Retail dataset available on the UCI Machine Learning Repository.

The dataset contains all transactions between 01/12/2009 and 09/12/2011 occurring for a UK-based non-store online retail company. The company mainly sells unique all-occasion gift-ware and many of its' customers are wholesalers.

def new_unique(df, group, date, period):
  unique = sales.resample(period, on=date, convention='end')[group].nunique()\
          .reset_index().set_index(date)
  min = sales.groupby(group)[date].min().reset_index()\
        .resample(period, on=date)[group].count().cumsum()
  nu = pd.merge(unique, min, on=date)
  nu.columns = ['Unique_in_Month', 'Unique_Cumulative']
  nu.plot(figsize=(10,5))
  plt.fill_between(nu.index, 0, nu.Unique_in_Month, alpha= 0.3)
  plt.fill_between(nu.index, nu.Unique_in_Month, nu.Unique_Cumulative, alpha= 0.3)
  plt.title('Unique monthly and cumulative ' + group + "'s")
  plt.ylabel('Unique ' + group + "'s") 
  plt.xlabel('Month, Year')
  plt.show()

def prop_plot(df, yr, date, group, value):
  df2 = df[df[date].dt.year == yr].groupby(group)[value].sum()\
      .sort_values(ascending=False).reset_index()
  df2['cvalue'] = df2[value].cumsum()
  # note groupby has filtered out any NA's, thus divide by sum of main df
  tot_val = df[df[date].dt.year == yr][value].sum()
  df2['cvalue_prop'] = df2['cvalue'] / tot_val
  df2['count'] = np.arange(1,(len(df2)+1))
  df2['count_prop'] = df2['count'] / (len(df2)+1)
  df_prop = df2[['count_prop', 'cvalue_prop']].set_index('count_prop')
  df_prop.plot(figsize=(10,5), label = 'proportion')
  plt.fill_between(df_prop.index, 0, df_prop.cvalue_prop, alpha= 0.3)
  plt.title('Proportion Plot: ' + group + ' vs. ' + value + ' for ' + str(yr))
  plt.ylabel('Proportion of ' + value) 
  plt.xlabel('Proportion of ' + group + "'s")
  plt.ylim(0,1.01) 
  plt.xlim(0,1.01) 
  plt.show()
